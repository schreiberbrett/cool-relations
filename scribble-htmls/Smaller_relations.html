5 Smaller relations

<palindromeo>

<palindromeo> ::=
(defrel (palindromeo l)
  (conde ((== l '()))
         ((fresh (x y)
                 (appendo `(,x . ,y) `(,x) l)
                 (palindromeo y)))))
