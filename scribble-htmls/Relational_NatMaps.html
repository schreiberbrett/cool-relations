Merging is just the relational equivalent of (zip-with conso) <relational-natmaps-merge> ::=
(defrel (mergeo ml mr mo)
  (fresh (hl tl hr tr to)
         (conde ((== ml '()) (== mr '()) (== mo '()))
                ((== ml '()) (== mr `(,hr . ,tr)) (== mr mo))
                ((== ml `(,hl . ,tl)) (== mr '()) (== ml mo))
                ((== ml `(,hl . ,tl)) (== mr `(,hr . ,tr)) (== mo `((,hl ,hr) . ,to))
                                      (mergeo tl tr to)))))
